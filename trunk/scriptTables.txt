CREATE TABLE CONTROL_SEND (
    ID         INTEGER       PRIMARY KEY,
    DATAENVIO  DATE,
    CAMINHOARQ VARCHAR (300),
    EMAILENVIO VARCHAR (200),
    ENVIADO    CHAR (1)      DEFAULT N
);


#FIM

CREATE TABLE LOG (
    ID         INTEGER       PRIMARY KEY,
    ID_CONTROL INTEGER       REFERENCES CONTROL_SEND (ID) ON DELETE CASCADE
                                                          ON UPDATE CASCADE,
    DATAENVIO  DATETIME,
    MENSAGEM   VARCHAR (300) 
);

#FIM
